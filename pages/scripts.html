OpenWrt 
#!/bin/sh
uci set network.lan.proto='static'
uci set network.lan.ipaddr='192.168.1.1'
uci set network.lan.netmask='255.255.255.0'
uci commit network
/etc/init.d/network restart

# Tambah firewall rule
uci add firewall rule
uci set firewall.@rule[-1].name='Allow-SSH'
uci set firewall.@rule[-1].src='wan'
uci set firewall.@rule[-1].proto='tcp'
uci set firewall.@rule[-1].dest_port='22'
uci set firewall.@rule[-1].target='ACCEPT'
uci commit firewall
/etc/init.d/firewall restart

# Monitoring jaringan
opkg update
opkg install vnstat
/etc/init.d/vnstat enable
/etc/init.d/vnstat start

# Update firmware
sysupgrade -v http://downloads.openwrt.org/releases/[version]/targets/[target]/openwrt-[version]-[target]-squashfs-sysupgrade.bin
Modem 
# Script untuk reset modem
echo 'ATZ' > /dev/ttyUSB0
sleep 5
echo 'AT+CGDCONT=1,"IP","123.456.789.0"' > /dev/ttyUSB0
sleep 2
echo 'ATD*99#' > /dev/ttyUSB0

# Script untuk cek sinyal
echo 'AT+CSQ' > /dev/ttyUSB0

# Script untuk konfigurasi modem
echo 'AT+CMGF=1' > /dev/ttyUSB0
echo 'AT+CSCS="GSM"' > /dev/ttyUSB0
echo 'AT+CMGS="12345"' > /dev/ttyUSB0
sleep 1
echo -ne 'Message body\x1A' > /dev/ttyUSB0
Router 
# Script untuk mengubah kebijakan firewall
iptables -A INPUT -p tcp --dport 22 -j ACCEPT
iptables -A INPUT -j DROP

# NAT konfigurasi
iptables -t nat -A POSTROUTING -o eth0 -j MASQUERADE
iptables -A FORWARD -i eth1 -o eth0 -j ACCEPT
iptables -A FORWARD -i eth0 -o eth1 -m state --state RELATED,ESTABLISHED -j ACCEPT

# Membuat backup konfigurasi
tar -cvzf /root/backup_$(date +%Y%m%d).tar.gz /etc/

# Scheduling otomatis restart
echo "0 3 * * * /sbin/reboot" >> /etc/crontabs/root

# Update firmware router
sysupgrade -v http://downloads.router-firmware.org/releases/[version]/targets/[target]/router-[version]-[target]-squashfs-sysupgrade.bin
